A real config — not a screenshot

One YAML, end-to-end run

A SFT → DPO run with safety eval, benchmark gate, and Annex IV export.

configs/policy-bot.yaml 
model:
  name_or_path: "Qwen/Qwen2.5-7B-Instruct"
  load_in_4bit: true
  max_length: 4096

lora:
  r: 16
  alpha: 32
  use_dora: true

datasets:
  - path: "data/policies.jsonl"
    format: "messages"

training:
  trainer: "sft"          # or dpo / simpo / kto / orpo / grpo
  epochs: 3
  batch_size: 2
  learning_rate: 2.0e-5

evaluation:
  benchmark:
    tasks: ["hellaswag", "arc_easy", "truthfulqa"]
    floors: { hellaswag: 0.55, arc_easy: 0.70 }
  safety:
    enabled: true
    model: "meta-llama/Llama-Guard-3-8B"
    block_categories: ["S1", "S2", "S5", "S10"]

compliance:
  annex_iv: true
  data_audit_artifact: "./audit/data_audit_report.json"
  human_approval: true     # Article 14 oversight gate

output:
  dir: "./checkpoints/policy-bot"
  model_card: true
  webhook: "${SLACK_WEBHOOK}"
terminal 
$ forgelm --config configs/policy-bot.yaml --dry-run   # validate
$ forgelm --config configs/policy-bot.yaml --fit-check  # VRAM check
$ forgelm --config configs/policy-bot.yaml             # run
